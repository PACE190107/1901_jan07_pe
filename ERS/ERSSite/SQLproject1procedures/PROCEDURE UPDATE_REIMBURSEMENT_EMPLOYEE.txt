create or replace PROCEDURE UPDATE_REIMBURSEMENT_EMPLOYEE 
(
  P_E_ID IN NUMBER 
, P_DATE_RECEIVED IN DATE 
, P_TYPE IN VARCHAR2
,P_AMOUNT IN NUMBER
,P_DESCRIPTION IN VARCHAR2
) AS 
BEGIN
UPDATE REIMBURSEMENTS SET AMOUNT = P_AMOUNT, DESCRIPTION = P_DESCRIPTION WHERE
E_ID = P_E_ID AND DATE_RECEIVED = P_DATE_RECEIVED AND TYPE = P_TYPE;

  
END UPDATE_REIMBURSEMENT_EMPLOYEE;
