CREATE OR REPLACE PROCEDURE UPDATE_REIMBURSEMENT_MANAGER 
(
  P_E_ID IN NUMBER 
, P_DATE_RECEIVED IN DATE 
, P_TYPE IN VARCHAR2 
,P_STATUS IN VARCHAR2
,P_MANAGER_ID IN NUMBER
) AS 
BEGIN
UPDATE REIMBURSEMENTS SET STATUS = P_STATUS, MANAGER_ID = P_MANAGER_ID, DATE_RESOLVED = SYSDATE WHERE
E_ID = P_E_ID AND DATE_RECEIVED = P_DATE_RECEIVED AND TYPE = P_TYPE;


END UPDATE_REIMBURSEMENT_MANAGER;
