create or replace PROCEDURE NEW_REIMBURSEMENT 
(
  P_E_ID IN NUMBER 
, P_DATE_RECEIVED IN DATE 
, P_DATE_RESOLVED IN DATE 
, P_TYPE IN VARCHAR2 
, P_AMOUNT IN NUMBER 
, P_DESCRIPTION IN VARCHAR2 
, P_STATUS IN VARCHAR2 
, P_MANAGER_ID IN NUMBER 
) AS
BEGIN 
INSERT INTO reimbursements (
E_ID,
DATE_RECEIVED,
DATE_RESOLVED,
TYPE,
AMOUNT,
DESCRIPTION,
STATUS, 
MANAGER_ID)
VALUES(
P_E_ID,
P_DATE_RECEIVED,
P_DATE_RESOLVED,
P_TYPE,
P_AMOUNT,
P_DESCRIPTION,
P_STATUS,
P_MANAGER_ID);


END NEW_REIMBURSEMENT;
